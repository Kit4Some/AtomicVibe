class CostTracker:
    async def track(self, model: str, input_tokens: int, output_tokens: int, purpose: str) -> None: ...

## 4. Execute Engine [Agent-E -> main.py, UI]

```python
class ExecuteEngine:
    def __init__(self, llm_router: LLMRouter, workspace_path: str): ...
    async def run(self, vibe_files: dict[str, str]) -> ExecuteStateV2: ...
    async def pause(self) -> None: ...
    async def resume(self) -> None: ...
    def get_status(self) -> dict: ...
    def get_diff(self) -> list[dict]: ...    # [{path, old, new, status}]
    def get_file_tree(self) -> list[dict]: ...
    def on_progress(self, callback: Callable) -> None: ...

class Supervisor:
    # Planner
    async def plan_sprint(self, state: ExecuteStateV2) -> dict: ...
    # Assigner
    async def assign_tasks(self, sprint_plan: dict, knowledge: list[dict]) -> list[dict]: ...
    # Reviewer
    async def review_code(self, code: dict, interfaces: str, conventions: str) -> ReviewResult: ...
    # Diagnostician
    async def diagnose(self, errors: list[dict], history: list[dict], knowledge: list[dict]) -> DiagnosisResult: ...
    # Strategist
    async def strategize(self, diagnosis: DiagnosisResult, context: dict) -> dict: ...
    # Retrospective
    async def retrospective(self, phase_data: dict) -> dict: ...

class Dispatcher:
    async def dispatch(self, agent_id: str, task_ids: list[int], vibe_files: dict, workspace_files: dict, injected_knowledge: list[str], errors: list | None) -> AgentCodeOutput: ...

class Validator:
    async def validate(self, workspace_path: str, phase: int) -> list[ValidationResult]: ...
    def all_passed(self, results: list[ValidationResult]) -> bool: ...

class Workspace:
    def write_files(self, files: list[CodeFile]) -> None: ...
    def read_file(self, path: str) -> str: ...
    def list_files(self, directory: str) -> list[str]: ...
    def git_commit(self, message: str) -> None: ...
    def git_rollback(self, tag: str) -> None: ...
    def git_tag(self, tag: str) -> None: ...
    def get_diff_since(self, tag: str) -> list[dict]: ...

class KnowledgeManager:
    def add(self, entry: KnowledgeEntry) -> None: ...
    def search(self, tags: list[str], limit: int = 5) -> list[KnowledgeEntry]: ...
    def update_confidence(self, entry_id: str, success: bool) -> None: ...
```
